int linearSearch(int arr[], int n, int key) {
    for (int i = 0; i < n; ++i) {
        if (arr[i] == key) {
            return i; // Return the index if key is found
        }
    }
    return -1; // Return -1 if key is not found
}
        
int binarySearch(const vector& arr, int key) {
    int low = 0;
    int high = arr.size() - 1;

    while (low <= high) {
        int mid = low + (high - low) / 2;

        if (arr[mid] == key) {
            return mid; // Return the index if key is found
        } 
        else if (arr[mid] < key) {
            low = mid + 1;
        } 
        else {
            high = mid - 1;
        }
    }

    return -1; // Return -1 if key is not found
}

int findFibonacci(int n) {
    int fibPrev = 0;
    int fibCurrent = 1;

    while (fibCurrent < n) {
        int temp = fibCurrent;
        fibCurrent += fibPrev;
        fibPrev = temp;
    }

    return fibCurrent;
}
// Fibonacci Search Function
int fibonacciSearch(int arr[], int n, int key) {
    int fibM2 = 0; // (m-2)th Fibonacci number
    int fibM1 = 1; // (m-1)th Fibonacci number
    int fib = fibM2 + fibM1; // mth Fibonacci number
    while (fib < n) {
        fibM2 = fibM1;
        fibM1 = fib;
        fib = fibM2 + fibM1;
    }
    int offset = -1;
    while (fib > 1) {
        int i = min(offset + fibM2, n - 1);

        if (arr[i] < key) {
            fib = fibM1;
            fibM1 = fibM2;
            fibM2 = fib - fibM1;
            offset = i;
        } 
        else if (arr[i] > key) {
            fib = fibM2;
            fibM1 -= fibM2;
            fibM2 = fib - fibM1;
        } 
        else {
            return i; // Return the index if key is found
        }
    }
    if (fibM1 && arr[offset + 1] == key) {
        return offset + 1; // Return the index if key is found
    }
    return -1; // Return -1 if key is not found
}
